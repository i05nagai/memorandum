#### Lemma 16

$$
\begin{eqnarray}
    \mathrm{E}_{w}

    \right]

        w_{0}
        +

    \label{equation_03_10}

    \mathrm{E}_{w}

    \right]

        }

            \hat{\xi}_{n}
            \hat{\xi}_{n}^{\mathrm{T}}
        }{
            n

$$

$$
\begin{eqnarray}

$$

        O_{p}(\frac{1}{n^{k/2}})
    \nonumber
    \\

        \left(
            \frac{\partial }{\partial \alpha} 
        \right)^{k}
        \mathcal{T}_{n}(\alpha)

    & = &

$$

#### proof

$$
\begin{eqnarray}
    f(x, w)^{k}

        f(x, w_{0})
        +
        \left(
            (w - w_{0})^{\mathrm{T}}
            \nabla f(x, w^{+})
        \right)^{k}
    \nonumber
    \\
\end{eqnarray}
$$
